#include<stdlib.h>
#include<math.h>

float* creer_matrice (int l, int c);
void afficher (float* m, int l, int c);
int colonne_pivot (float* m, int l, int c, int lc, int cp);
int ligne_pivot (float* m, int l, int c, int lc, int cp);

int main(int argc, char const *argv[]) {
  int l = 3, c = 5;
  float* m = creer_matrice(l,c);
  // initialisation
  m[0] = 1;m[1] = 0;m[2] = 7;m[3] = -3; m[4] = 5;
  m[5] = 0;m[6] = 3;m[7] = 6;m[8] = 0; m[9] = 3;
  m[10] = 2;m[11] = -2;m[12] = 0;m[13] = 4;m[14] = 8;

  afficher(m,l,c);

  descente(m,l,c);

  return 0;
}

float* creer_matrice (int l,int c){
  float* m = malloc(l*c*sizeof(float));
  for (int i = 0; i < l; i++) {
    for (int j = 0; j < c; j++) {
      m[i*c+j] = 0;
    }
  }
  return m;
}

void afficher (float* m, int l, int c){
  for (int i = 0; i < l; i++) {
    for (int j = 0; j < c; j++) {
      printf("%f\t", m[i*c+j]);
    }
    printf("\n");
  }
}

int colonne_pivot (float *m, int l, int c, int lc, int cp){
  if (c == cp){
    return c;
  } else {
    int i = lc;
    while (m[i*c+cp]==0 && i<l-1){
      i++;
    }
    if (m[i*c+cp] != 0)
      return cp;
    else
      return colonne_pivot(m,l,c,lc,cp+1);
  }
}

int ligne_pivot (float *m, int l, int c, int lc, int cp){
  int i;
  int lp = lc;
  float max = fabsf(m[lc*c+cp]);
  for (i = lc+1; i < l; i++) {
    if (m[i*c+cp] > max){
      lp = i;
    }
  }
  return lp;
}

void echange (float* m, int l, int c, int lc, int lp){
  float temp = 0.;
  for (int i = 0; i < c; i++) {
    temp = m[lc*c+i];
    m[lc*c+i] = m[lp*c+i];
    m[lp*c+i] = temp;
  }
}

void annule (float* m, int l, int c, int lc, int cp){
  for (int i = lc+1; i < l; i++) {
    float k = m[i*c+cp]/m[lc*c+cp];
    m[i*c+cp] = 0;
    for (int j = cp+1; j < c; j++) {
      m[i*c+j] = m[i*c+j] - k*m[lc*c+j];
    }
  }
}

void annule_normalise (float* m, int l, int c, int lc, int cp){
  float pivot = m[lc*c+cp];
  if(pivot!=0){
    m[lc*c+cp] = 1;
    for (int j = cp+1; j < c; j++){
      m[lc*c+j] /= pivot;
    }
    for (int i = 0; i < l; i++){
      if (i != lc){
        float k = m[i*c+cp];
        m[i*c+cp] = 0;
        for (int j = cp+1; j < c; j++){
          m[i*c+j] = m[i*c+j] - k*m[lc*c+j];
        }
      }
    }
  }
}

void descente(float* m, int l, int c){
  int lc = 0; int cp = 0; int lp =0;
  printf("ligne courante = %d\n", lc);
  while (lc < l){
    cp = colonne_pivot(m,l,c,lc,cp);
    if (cp == c-1)
      break;
    lp = ligne_pivot(m,l,c,lc,cp);
    if (lp != lc)
      echange(m,l,c,lc,lp);
    //annule(m,l,c,lc,cp);
    annule_normalise(m,l,c,lc,cp);
    lc++;
    printf("ligne courante = %d\n", lc);
    afficher(m,l,c);
  }
}

#include <stdio.h>
#include <stdlib.h>

void affiche_tab(int* tab,int n);
void affiche_matrice(int *M, int n);
int produit_min(int* indices, int taille, int j, int k, int* memoire);

int main(int argc, char const *argv[]) {
  if (argc < 2)
    printf("Trop peu d'arguments: mettre en argument la taille de la chaine de matrices\n");
  else {

    // On récupère la longueur de la chaine de matrice
    int n = atoi(argv[1]);
    // La taille du tableau d'indices
    int taille = n+1;

    // Le tableau d'indices i_0 ... i_n contient donc n+1 elements
    // Si on veut que la suite soit 1 2 3 .... n
    // Pour une chaine de n matrice on aura comme suite d'indices
    // 1 2 3 4 ... n+1
    int* indices = malloc((n+1)*sizeof(int));
    for (int i = 0; i <= n; i++) {
      indices[i] = i+1;
    }
    printf("La suite d'indices vaut:\n");
    affiche_tab(indices,n);

    // On alloue la matrice où on stockera les valeurs de calul intermédiaire
    // Elle est de taille (n+1) * (n+1)
    // Et sera initialisé à 0
    int* memoire = calloc((n+1)*(n+1),sizeof(int));

    // On calcule ensuite le nombre de produit minimal pour le sous-problème mu_0n
    // C'est à dire pour mu[i_0][i_n] qui valent ici mu[1][n+1]
    int mu = produit_min(indices,taille,0,n,memoire);
    printf("--\n");
    printf("La matrice memoire:\n");
    affiche_matrice(memoire,n+1);
    printf("--\n");
    printf("Le nombre de produits minimal pour une chaine de matrice de longueur %d est %d\n",n,mu);
  }

  return 0;
}


int produit_min(int* indices, int taille, int j, int k, int* memoire) {
  // cas terminal
  if (j+1 == k) return 0;

  // Variables temporaires
  int mu_jk, mu_jc, mu_ck;
  // L'indice c où on effectue le découpage
  int c;
  // On initialise la valeur de mu_jk à -1
  mu_jk = -1;
  // Pour j<c<k on résoud les sous-problème si ce n'est pas encore fait
  for (c = j+1; c < k; c++) {
    if (memoire[j*taille+c] == 0) {
      memoire[j*taille+c] = produit_min(indices,taille,j,c,memoire);
    }
    mu_jc = memoire[j*taille+c];
    if (memoire[c*taille+k] == 0) {
      memoire[c*taille+k] = produit_min(indices,taille,c,k,memoire);
    }
    mu_ck = memoire[c*taille+k];

    // On stocke temporairement la valeur de la solution optimale mu_jk
    int min_temp = mu_jc + mu_ck + indices[j]*indices[c]*indices[k];
    // Si elle n'a pas été modifiée ou si l'on trouve une meilleure solution
    // On la met à jour
    if (mu_jk<0 || min_temp < mu_jk) {
      mu_jk = min_temp;
    }
  }
  memoire[j*taille+k] = mu_jk;
  return mu_jk;
}

void affiche_tab(int* tab,int n) {
  for (int i = 0; i <= n; i++) {
    printf("%d ",tab[i]);
  }
  printf("\n");
}

void affiche_matrice(int *M, int n) {
  for (size_t i = 0; i < n; i++) {
    for (size_t j = 0; j < n; j++) {
      printf("%d ", M[i*n+j]);
    }
    printf("\n");
  }
}
