#include<stdio.h>
#include<stdlib.h>
#include<time.h>

#include"liste.h"


liste_t* initialiser_liste() {
  liste_t* l = malloc(sizeof(liste_t));
  l->debut = NULL;
  l->taille = 0;
  return l;
}

maillon_t* creer_maillon(int val) {
  maillon_t* res = malloc(sizeof(maillon_t));
  res->val = val;
  res->suivant = NULL;
  return res;
}

void maillon_detruire(maillon_t* m) {
  free(m);
}

void liste_detruire(liste_t* l){
  for (size_t i = 0; i < l->taille; i++) {
    maillon_detruire(extraire_debut(l));
  }
  free(l);
}

void affiche_liste(liste_t l) {
  maillon_t* m = l.debut;
  while(m != NULL){
    printf("%d ",m->val);
    m = m->suivant;
  }
  printf("\n");
}

void ajouter_debut(liste_t* l, maillon_t* m) {
  m->suivant = l->debut;
  l->debut = m;
  l->taille++;
}

maillon_t* extraire_debut(liste_t* l) {
  maillon_t* m = l->debut;
  if (m != NULL){
    l->debut = m->suivant;
    m->suivant = NULL;
    l->taille--;
  }
  return m;
}

liste_t* creer_liste_circulaire(int n) {
  liste_t* l = malloc(sizeof(liste_t));
  //on cree le premier maillon
  maillon_t* m = creer_maillon(1);
  l->debut = m;
  l->taille = n;

  m->suivant = m;
  maillon_t* temp = m;

  for (size_t i = 2; i <= n; i++) {
    temp = (temp->suivant = creer_maillon(i));
    temp->suivant = m;
  }
  return l;
}

maillon_t* election(liste_t* l, int m) {
  maillon_t* cur = l->debut;
  maillon_t* temp;
  printf("Valeurs détruite: ");
  while (cur != cur->suivant){
    for (size_t i = 1; i < m; i++)
      cur = cur ->suivant;
    temp = cur->suivant;
    cur->suivant = cur->suivant->suivant;
    printf("%d ", temp->val);
    maillon_detruire(temp);
  }
  printf("\n");
  return cur;
}

liste_t* creer_aleatoire(int n) {
  liste_t* l = initialiser_liste();
  maillon_t* temp;
  srand(time(NULL));
  for (size_t i = 0; i < n; i++) {
    temp = creer_maillon(rand()%100);
    ajouter_debut(l,temp);
  }
  return l;
}

// Cette fonction prend une liste déjà triée et insère un maillon à sa bonne place
void inserer_dans_trie(liste_t* l, maillon_t* nouveau) {
  // On créer deux curseur prec et cur (l'idée est que prec->suiv = cur)
  // Et on cherchera à insérer le nouveau maillon à la bonne place dans une liste déjà triée
  maillon_t* prec;
  maillon_t* cur;
  cur = l->debut;
  //si on a une liste vide ou que la valeur du nouveau maillon est plus petite que celui du debut de la liste
  if (cur == NULL || cur->val >= nouveau->val) {
    ajouter_debut(l,nouveau);
  }
  // sinon
  else {
    // tant qu'on n'atteint pas la fin de la liste et que la valeur du curseur est plus petite que celle du nouveau maillon, on va déplacer le curseur
    while (cur != NULL && cur->val < nouveau->val) {
      prec = cur;
      cur = cur->suivant;
    }
    nouveau->suivant = cur;
    prec->suivant = nouveau;
    l->taille++;
  }
}

// Cette fonction prend une liste quelconque et renvoie une nouvelle liste avec les mêmes valeurs mais cette fois-ci triée
liste_t* tri_liste(liste_t* l) {
  liste_t* tri = initialiser_liste();
  // tant que la liste l n'est pas vide, on insère dans res le maillon "début" de la liste l avec la fonction inserer_dans_trie
  while (l->debut != NULL) {
    maillon_t* nouveau = extraire_debut(l);
    inserer_dans_trie(tri,nouveau);
  }
  return tri;
}

#include<stdio.h>
#include<stdlib.h>

#include"liste.h"


int main(int argc, char const *argv[]) {
if (argc < 2) printf("Trop peu d'argument\n");
else {
  int n = atoi(argv[1]);

  printf("Une liste aleatoire de taille %d:\n",n);
  liste_t* l = creer_aleatoire(n);
  affiche_liste(*l);

  printf("Une nouvelle liste avec les memes valeurs triees:\n");
  liste_t* tri = tri_liste(l);
  affiche_liste(*tri);

  liste_detruire(l);
  liste_detruire(tri);
}
return 0;
}
