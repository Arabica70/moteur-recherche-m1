    if tab == []:
        return []
    else:  # concatenation de
        # 1. retourne appliquee a tab sans le 1er element
        # 2. le premier element de tab
        return retourne(tab[1:]) + tab[0]

2.4 Tri rapide \\ listes en compréhension

def tri_rapide(tab):
    if tab == []:
        return []
    else:
        pivot = tab[0]
        # on selectionne les elements plus petits que pivot
        pp = [x for x in tab if x < pivot]
        # on selectionne les elements plus grands que pivot
        pg = [x for x in tab if x > pivot]
        return tri_rapide(pp) + [pivot] + tri_rapide(pg)

3.1 Lancer un programme python

3 façons :

  1. Utiliser l'interpréteur
  2. Dans un fichier foo.py
    • lancer la commande
      • python foo.py
  3. Faire un exécutable
    • ajouter la première ligne
      • #! /usr/bin/python
    • rendre foo.py exécutable

    • le lancer directement

3.9 Expressions régulières

# pat : expression reguliere
# str : chaine de la recherche
# retourne la premiere occurrence trouvee
match = re.search(pat, str)

str = 'an example word:cat!!'
match = re.search(r'word:\w\w\w', str)
if match:
    print 'found', match.group() ## 'found word:cat'

3.10 Expressions régulières (2)

## Si trouve, match.group() est le texte trouve
match = re.search(r'iii', 'piiig')
# => trouve, match.group() == "iii"
match = re.search(r'igs', 'piiig')
# =>  pas trouve, match == None

## . = tout caractere sauf  \n
match = re.search(r'..g', 'piiig')
# =>  trouve, match.group() == "iig"

## \d = chiffre, \w = lettre
match = re.search(r'\d\d\d', 'p123g')
# =>  trouve, match.group() == "123"
match = re.search(r'\w\w\w', '@@abcd!!')
# =>  trouve, match.group() == "abc"

3.12 Expressions régulières (4) \\ groupement

str = 'purple alice-b@google.com monkey dishwasher'
match = re.search('([\w.-]+)@([\w.-]+)', str)
if match:
  print match.group()   # 'alice-b@google.com' (le match)
  print match.group(1)  # 'alice-b' (le login, group 1)
  print match.group(2)  # 'google.com' (le serveur, group 2)
